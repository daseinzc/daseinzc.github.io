Stata模拟中心极限定理

关于中心极限定理的模拟

do文件如下：

clear
set seed 101   // 让我给试出个好看的图
set obs 50000  // 设置 50000 个观测值，用于存储均值

* 预先生成 group_means 变量，用于存储每次计算的均值
generate group_means = .

capture frame drop temp_frame
frame create temp_frame 

* 循环生成样本并计算均值
forvalues i = 1/50000 {
    frame temp_frame{
        clear     // 新建一个frame
        
        quietly{
            set obs 50 //样本数
            generate x = runiformint(1,6)  // 每个样本包含 50 个 1 到 6 的随机整数

            * 计算样本均值
            summarize x, meanonly
            
            scalar temp_mean = r(mean)  // 将均值保存到标量 temp_mean
        }
    }
    
    quietly replace group_means = temp_mean in `i'
    
    if mod(`i', 2500) == 0 {
        display "Progress: " %5.2f ((`i'/50000)*100) "% complete"
    }
}

frame drop temp_frame //删除临时框架 

* 绘制 group_means 的直方图叠加正态分布曲线
histogram group_means, start(1) normal ///
xscale(range(1 6)) xlabel(1(1)6)
